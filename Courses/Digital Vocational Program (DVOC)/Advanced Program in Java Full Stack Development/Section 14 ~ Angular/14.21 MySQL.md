## **Step 1: Setting Up MySQL Database**

1. **Open MySQL Workbench**:
    
    - **Create the Database**:
        
    
    ```sql
    CREATE DATABASE CustomerRecords;
    ```
    
2. **Select the Database**:
    
    ```sql
    USE CustomerRecords;
    ```
    
3. **Create `customer` Table**:
    
    ```sql
    CREATE TABLE customer (
      custid INT PRIMARY KEY,
      firstname VARCHAR(45),
      lastname VARCHAR(45),
      email VARCHAR(100),
      cellno VARCHAR(10)
    );
    ```
    
4. **Fix Authentication Error** (If encountered):
    
    - Error: `ER_NOT_SUPPORTED_AUTH_MODE`
        
    - **Solution**: Update MySQL authentication method:
        
    
    ```sql
    ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password BY 'your_password';
    ```
    
5. **Insert Sample Data**:
    

## **Step 2: Set Up Node.js Backend**

1. **Create Project Directory and Initialize Node.js Project**:
    
    ```bash
    npm init
    ```
    
2. **Install Required Packages**:
    
    ```bash
    npm install express body-parser mysql cors
    ```
    
3. **Create `server.js` File**:
    
    ```js
    const express = require('express');
    const bodyParser = require('body-parser');
    const mysql = require('mysql');
    const cors = require('cors');
    const server = express();
    
    const db = mysql.createConnection({
        host: "localhost",
        user: "root",
        password: "your_password",  // Update with your MySQL password
        database: "customerrecords",
        port: 3306
    });
    
    // CORS setup for frontend (Angular)
    server.use(cors({
        origin: 'http://localhost:4200',
        methods: ['GET', 'POST', 'DELETE', 'PUT'],
        allowedHeaders: ['Content-Type']
    }));
    
    server.use(express.json());  // Middleware for JSON parsing
    server.use(bodyParser.json());
    
    // Check DB connection
    db.connect(function(err) {
        if (err) {
            console.log("Can't connect to the database!");
            console.log(err);
        } else {
            console.log("Successfully connected to database!");
        }
    });
    
    // Start server on port 8089
    server.listen(8089, function(err) {
        if (err) {
            console.log("Cannot connect to the server!");
            console.log(err);
        } else {
            console.log("Successfully connected to server at port 8089!");
        }
    });
    
    // Get all customers
    server.get('/api/customer/getall', (req, res) => {
        const query = "SELECT * FROM customer";
        db.query(query, (error, result) => {
            if (error) {
                res.send("Error: " + error);
            } else {
                res.send({ status: "OK", data: result });
            }
        });
    });
    
    // Fetch customer by ID
    server.get('/api/customer/get/:id', (req, res) => {
        const customerId = req.params.id;
        const query = "SELECT * FROM customer WHERE custid = ?";
    
        db.query(query, [customerId], (error, result) => {
            if (error) {
                res.status(500).send({ status: false, message: "Error retrieving customer!" });
                console.log(error);
            } else {
                if (result.length > 0) {
                    res.status(200).send({ status: "OK", data: result });
                } else {
                    res.status(404).send({ status: false, message: "Customer not found!" });
                }
            }
        });
    });
    
    // Save customer data
    server.post('/api/customer/save', (req, res) => {
        const customerData = req.body;
        const query = "INSERT INTO customer (custid, firstname, lastname, email, cellno) VALUES (?, ?, ?, ?, ?)";
        
        db.query(query, [customerData.custid, customerData.firstname, customerData.lastname, customerData.email, customerData.cellno], (error) => {
            if (error) {
                res.status(500).send({ status: false, message: "Customer creation failed!" });
                console.log(error);
            } else {
                res.status(200).send({ status: true, message: "Customer created successfully!" });
            }
        });
    });
    
    // Delete customer by ID
    server.delete("/api/customer/delete/:id", (req, res) => {
        const customerId = req.params.id;
        const query = "DELETE FROM customer WHERE custid = ?";
    
        db.query(query, [customerId], (err, result) => {
            if (err) {
                res.status(500).send({ status: false, message: "Customer deletion failed!" });
                console.log(err);
            } else {
                if (result.affectedRows > 0) {
                    res.status(200).send({ status: true, message: "Customer deleted successfully!" });
                } else {
                    res.status(404).send({ status: false, message: "Customer not found!" });
                }
            }
        });
    });
    
    // Update customer data
    server.put("/api/customer/edit/:id", (req, res) => {
        const updateQuery = "UPDATE customer SET firstname = ?, lastname = ?, email = ?, cellno = ? WHERE custid = ?";
        const values = [req.body.firstname, req.body.lastname, req.body.email, req.body.cellno, req.params.id];
    
        db.query(updateQuery, values, (error) => {
            if (error) {
                res.status(500).send({ status: false, message: "Customer update failed!" });
                console.log(error);
            } else {
                res.status(200).send({ status: true, message: "Customer updated successfully!" });
            }
        });
    });
    ```

## **Step 3: Set Up Angular Project**

1. **Create Angular Project**:
    
    ```bash
    ng new mysql-crud
    ```
    
2. **Generate Components**:
    
    ```bash
    ng generate component create
    ng generate component retrieve
    ng generate component update
    ```
    
3. **Generate Customer Service**:
    
    ```bash
    ng generate service customer
    ```
    
4. **Set Up Routing**: Update `app-routing.module.ts` to configure routes for your components:
    
    ```ts
    import { NgModule } from '@angular/core';
    import { RouterModule, Routes } from '@angular/router';
    import { CreateComponent } from './create/create.component';
    import { RetrieveComponent } from './retrieve/retrieve.component';
    import { UpdateComponent } from './update/update.component';
    
    const routes: Routes = [
      { path: "create", component: CreateComponent },
      { path: "retrieve", component: RetrieveComponent },
      { path: "update/:id", component: UpdateComponent }
    ];
    
    @NgModule({
      imports: [RouterModule.forRoot(routes)],
      exports: [RouterModule]
    })
    export class AppRoutingModule { }
    ```

## **Customer Service (customer.service.ts)**

```ts
import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';

@Injectable({
  providedIn: 'root'
})
export class CustomerService {
  private baseUrl = "http://localhost:8089";  // Backend API URL

  constructor(private httpRouter: HttpClient) { }

  getAllCustomers() {
    return this.httpRouter.get(this.baseUrl + "/api/customer/getall");
  }

  saveCustomer(custData: any) {
    return this.httpRouter.post(this.baseUrl + "/api/customer/save", custData);
  }

  deleteCustomer(id: any) {
    return this.httpRouter.delete(this.baseUrl + "/api/customer/delete/" + id);
  }

  updateCustomer(id: any, body: any) {
    return this.httpRouter.put(this.baseUrl + "/api/customer/edit/" + id, body);
  }

  getCustomerById(id: any) {
    return this.httpRouter.get(this.baseUrl + "/api/customer/get/" + id);
  }
}
```

### **Retrieve Component**

#### **retrieve.component.ts**

```ts
import { Component } from '@angular/core';
import { CustomerService } from '../services/customer.service';

@Component({
  selector: 'app-retrieve',
  templateUrl: './retrieve.component.html',
  styleUrls: ['./retrieve.component.css']
})
export class RetrieveComponent {
  customers: any[] = [];

  constructor(private customerService: CustomerService) {}

  ngOnInit() {
    this.customerService.getAllCustomers().subscribe(data => {
      this.customers = data['data'];
    });
  }

  deleteCustomer(id: any) {
    this.customerService.deleteCustomer(id).subscribe(() => {
      this.customers = this.customers.filter(c => c.custid !== id);
    });
  }
}
```

#### **retrieve.component.html**

```html

    <td>{{ customer.firstname }} {{ customer.lastname }}</td>
```

### **Update Component**

#### **update.component.ts**

```ts
import { Component } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { CustomerService } from '../services/customer.service';

@Component({
  selector: 'app-update',
  templateUrl: './update.component.html',
  styleUrls: ['./update.component.css']
})
export class UpdateComponent {
  customerData: any = {};

  constructor(
    private customerService: CustomerService,
    private route: ActivatedRoute,
    private router: Router
  ) {}

  ngOnInit() {
    const id = this.route.snapshot.paramMap.get('id');
    this.customerService.getCustomerById(id).subscribe((data) => {
      this.customerData = data['data'][0];
    });
  }

  updateCustomer() {
    this.customerService.updateCustomer(this.customerData.custid, this.customerData).subscribe(() => {
      this.router.navigate(['/retrieve']);
    });
  }
}
```
